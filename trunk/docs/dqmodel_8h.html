Declare a model.

For example,

#include <dquest.h>

class User : public DQModel {
    DQ_MODEL
public:
    DQField<QString> userId;
    DQField<QDateTime> creationDate;
    DQField<qreal> karma;
};

DQ_DECLARE_MODEL(User,
                 "user", // the table name.
                 DQ_FIELD(userId , DQNotNull | DQUnique),
                 DQ_FIELD(creationDate , DQDefault("CURRENT_TIMESTAMP") ),
                 DQ_FIELD(karma)
                 );
